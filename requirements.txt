# Entity resolution
# ---------------------------------------------------------------------------
splink==4.0.16

# DO NOT BUMP CASUALLY.
#
# History: a "Salting partitions must be specified" failure from
# estimate_u_using_random_sampling() was previously attributed to this pairing,
# and the planned workaround was to skip random-sampling u-estimation entirely.
# RETESTED on splink 4.0.16 + duckdb 1.5.4 (Windows, Python 3.12): it does NOT
# reproduce - not with the full comparison set, not with max_pairs up to 1e7,
# not with an explicit seed. src/tools/entity_resolution.py therefore uses the
# standard training path (deterministic rules -> lambda, random sampling -> u,
# EM -> m), which yields more reliable u values than backing them out of EM.
#
# If the salting error resurfaces on another platform, the fallback is to drop
# estimate_u_using_random_sampling() and call
# estimate_parameters_using_expectation_maximisation(fix_u_probabilities=False).
# Do NOT downgrade duckdb to 1.4.0 / 1.3.2 to dodge it - that path introduces
# other type-incompatibility problems.
duckdb==1.5.4

# ---------------------------------------------------------------------------
# Semantic matching (field mapping suggestions)
# ---------------------------------------------------------------------------
sentence-transformers==5.6.0

# Pin is 2.13.0, NOT 2.13.0+cpu. The "+cpu" suffix is a local-version tag that
# only exists on download.pytorch.org; PyPI publishes plain 2.13.0, so pinning
# "+cpu" fails with "No matching distribution found". The Windows PyPI wheel is
# already CPU-only (torch.__version__ then reports 2.13.0+cpu, which is normal).
# For an explicit CPU build on Linux:
#   pip install torch==2.13.0 --index-url https://download.pytorch.org/whl/cpu
torch==2.13.0

rapidfuzz==3.14.5
